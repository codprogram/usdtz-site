  <script>

    // =============================

    // CONFIG — EDITE SOMENTE AQUI

    // =============================

    const CONFIG = {

      name: "USDT.Z",

      symbol: "USDT.Z",

      decimals: 18,

      address: "0x5f23aA7056562B1d0Fa2AB4A55890CbB02A4958d",

      image: "https://raw.githubusercontent.com/codprogram/programacaoC/refs/heads/main/usdtz_logo_transparente.png",

      networkName: "BNB Smart Chain (Mainnet 56)",

      explorerUrl: "https://bscscan.com/token/0x5f23aA7056562B1d0Fa2AB4A55890CbB02A4958d",

      metamaskType: "ERC20",

      trustWalletAsset: "c20000714_t0x5f23aA7056562B1d0Fa2AB4A55890CbB02A4958d"

    };


    function setText(id, value) {

      document.getElementById(id).textContent = value;

    }


    function applyConfig() {

      setText("tokenName", CONFIG.name);

      setText("tokenNameInfo", CONFIG.name);

      setText("tokenSymbolBadge", CONFIG.symbol);

      setText("tokenSymbolInfo", CONFIG.symbol);

      setText("tokenAddressInfo", CONFIG.address);

      setText("tokenNetworkBadge", CONFIG.networkName);

      setText("tokenNetworkInfo", CONFIG.networkName);

      setText("tokenDecimalsBadge", CONFIG.decimals);

      setText("tokenDecimalsInfo", CONFIG.decimals);


      document.getElementById("tokenImagePreview").src = CONFIG.image;

      document.getElementById("explorerLinkText").href = CONFIG.explorerUrl;

      document.getElementById("explorerLinkBtn").href = CONFIG.explorerUrl;

      document.getElementById("trustWalletBtn").href = `https://link.trustwallet.com/add_asset?asset=${encodeURIComponent(CONFIG.trustWalletAsset)}`;

    }


    function setStatus(message) {

      document.getElementById("statusBox").textContent = message;

    }


    async function copyContract() {

      try {

        await navigator.clipboard.writeText(CONFIG.address);

        setStatus("Contrato copiado com sucesso.");

      } catch (error) {

        setStatus("Não foi possível copiar automaticamente. Copie manualmente o contrato exibido.");

      }

    }


    async function addToMetaMask() {

      if (!window.ethereum) {

        setStatus("MetaMask não detectada neste navegador.");

        return;

      }


      try {

        const wasAdded = await window.ethereum.request({

          method: "wallet_watchAsset",

          params: {

            type: CONFIG.metamaskType,

            options: {

              address: CONFIG.address,

              symbol: CONFIG.symbol,

              decimals: Number(CONFIG.decimals),

              image: CONFIG.image

            }

          }

        });


        if (wasAdded) {

          setStatus("Token enviado para confirmação e adicionado na MetaMask.");

        } else {

          setStatus("A solicitação foi cancelada pelo usuário.");

        }

      } catch (error) {

        console.error(error);

        setStatus("Erro ao tentar adicionar o token na MetaMask.");

      }

    }


    document.getElementById("copyContractBtn").addEventListener("click", copyContract);

    document.getElementById("addMetaMaskBtn").addEventListener("click", addToMetaMask);


    applyConfig();

  </script>
